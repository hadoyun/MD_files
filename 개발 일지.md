# 개발 일지



```cpp
#include <vector>
#include <iostream>

using namespace std;

enum class e_DayOfWeek
{
	월,
	화,
	수,
	목,
	금,
	토,
	일
};


//마비노기 쿠폰 계산기!

int Coupon(e_DayOfWeek DOF, int days, int current_bonus)
{
	for (int i = 0; i < days; ++i)
	{
		current_bonus += 1;

		int id{ static_cast<int>(DOF) + 1 };
        
		if (id == 7) id = 0;

		DOF = static_cast<e_DayOfWeek>(id);

		if (DOF == e_DayOfWeek::일 || DOF == e_DayOfWeek::토)
		{
			current_bonus += 1;
		}

	}

	return current_bonus;
}


int main()
{

	int z{};
	
	z = Coupon(e_DayOfWeek::목, 14, 1);

	cout << z;

	return 0;
}
```
